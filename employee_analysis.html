# employee_analysis.py

import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

# Generate sample data
np.random.seed(0)
departments = ["Sales", "Engineering", "HR", "Marketing", "Operations", "Finance"]
regions = ["North", "South", "East", "West"]

df = pd.DataFrame({
    "EmployeeID": range(1, 101),
    "Department": np.random.choice(departments, 100),
    "Region": np.random.choice(regions, 100),
    "PerformanceScore": np.random.randint(50, 101, 100)
})

# Frequency count for Sales department
sales_count = df[df["Department"] == "Sales"].shape[0]
print(f"Number of employees in Sales department: {sales_count}")

# Histogram
sns.set_style("whitegrid")
plt.figure(figsize=(8,6))
sns.countplot(data=df, x="Department", palette="Set2")
plt.title("Employee Distribution by Department")
plt.xlabel("Department")
plt.ylabel("Count")
plt.tight_layout()
plt.show()
